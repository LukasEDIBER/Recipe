\newcommand{\recipe}[2][]{%
    \newpage
    \lfoot{}%
    \rfoot{}%
    \section[\small #2]{\huge #2}%
    \if###1##%
    \else
        \begin{center}
            \parbox{0.75\linewidth}{\normalfont\raggedright\itshape#1}%
        \end{center}
        \noindent\rule{\textwidth}{0.4pt}
    \fi
}
\newcommand{\showit}[3][1in]{%
    \begin{center}
            \includegraphics[width=2in,height=2in,keepaspectratio]{#2}%
            \par
            %\medskip
            %\emph{#3}
            \par
    \end{center}%
    }

%% Optional argument for a  heading within the ingredients section
\newcommand{\ingredients}[1][]{%
    \if###1##%
        {}%
\else
    \medskip
        \emph{\normalsize\textbf{#1}}%
    \fi
}

%% Use \obeylines to minimize markup
\newenvironment{ingreds}{%
    
    \parindent0pt
    \noindent
    \ingredients
    \par
    \smallskip
    \footnotesize
    \begin{multicols}{2}
    \leftskip0em
    \rightskip0pt plus 0em
    \parskip=0.20em
    \obeylines
    \everypar={\hangindent2em}
    
}{%
    \end{multicols}%
    \medskip
}

%% Optional argument for an italicized pre-step

\newenvironment{method}[1][]{%
    \medskip
    \noindent
    {\itshape\normalsize\textbf{Zubereitung}}%
    \par
    \smallskip
    \if###1##%
    \else
        \noindent
        \emph{#1}
        \par
    \fi
    \begingroup
    \parindent0pt
    \parskip0.25em
        \leftskip0em
    
}{%
    \par
    \endgroup}
